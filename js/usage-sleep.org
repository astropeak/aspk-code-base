* sleep
  Sleep implemented by Promise. Looks quite good!
  
  #+begin_src js
  function sleep (time) {
      return new Promise((resolve) => setTimeout(resolve, time));
  }
  
  // Usage!
  sleep(500).then(() => {
      // Do something after the sleep!
  })
  #+end_src
